Remarks

This extension methods is intented to be used via LINQ syntax.

Examples
var m = from v1 in new SimpleMonad<int>(3)
        from v2 in new SimpleMonad<int>(v1*2)
        select v1 + v2;
yield return m.Do();
Assert.That( m.Result == 9 );
